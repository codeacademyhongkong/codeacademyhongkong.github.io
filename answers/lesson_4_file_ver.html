# Question 1
count = 1
while count <= 50:
    count_str = str(count)
    if '3' in count_str or '6' in count_str or '9' in count_str:
        print('*claps*')
    else:
        print(count_str)
    count = count + 1


# Question 2
names = ['Alex', 'Sonia', 'Julia', 'Leona', 'Johan']
boys = []
girls = []

for name in names:
    if name[-1] == 'a':
        girls.append(name)
    else:
        boys.append(name)

print('Boys: ', boys)
print('Girls: ', girls)

# while loop
fibonacci_numbers = []
previous = 0
current = 1
index = 1 # as the initial value for current is the 1st 
while index < 50: 
    index += 1 # equiv to index = index + 1
    temp = current
    current = previous + current
    previous = temp
    fibonacci_numbers.append(current)
print(fibonacci_numbers)

# for loop
fibonacci_numbers = []
previous = 0
current = 1
for x in range(50 - 1):
    previous, current = current, previous + current
    fibonacci_numbers.append(current)
print(fibonacci_numbers)
