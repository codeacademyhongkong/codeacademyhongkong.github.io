# Question 1

    'name': 'Johan',
    'length': 180
}
a = {
    'name': 'Alex',
    'length': 190
}

total_length = j['length'] + a['length']
people = {
    'data': [j, a],
    'total_length': total_length
}

print(people)



# Question 2
cities = [
    {'city': 'HK', 'size': 7}, 
    {'city': 'NY', 'size': 10}
]

flipped = {
    'names': [],
    'sizes': []
}

for city in cities:
    city_name = city['city']
    city_size = city['size']

    flipped['names'].append(city_name)
    flipped['sizes'].append(city_size)

print(flipped)
