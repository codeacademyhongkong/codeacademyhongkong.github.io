# Question 1
import requests

def get_page_snippet(url, first_x_chars=1000):
    result = requests.get(url)
    body = result.text[:first_x_chars]
    return body

hcom = get_page_snippet('https://zh.hotels.com')
print(hcom)

bex = get_page_snippet('https://www.expedia.com/', first_x_chars=2500)
print(bex)

google = get_page_snippet('https://www.google.com/', first_x_chars=10000)
print(google)
